#!/bin/bash

# Open a new tmux session named 'session1' and execute test1.py
tmux new-session -d -s session1 "python test.py"

# Detach from 'session1'
tmux detach-client

# Open a new tmux session named 'session2' and execute test2.py
tmux new-session -d -s session2 "python test2.py"

# Detach from 'session2'
tmux detach-client

# Print 'hello'
echo "hello"
